import express from "express"; import fetch from "node-fetch"; const app = express(); app.get("/proxy/m3u", async (req, res) => { const target = req.query.url; if (!target) return res.status(400).send("URL yoxdur"); try { const r = await fetch(target); const data = await r.text(); res.setHeader("Content-Type", "application/vnd.apple.mpegurl"); res.send(data); } catch (err) { res.status(500).send("Xəta: " + err.message); } }); app.listen(3000, () => console.log("Proxy işləyir"));
